        function calcGrade(i) {
            var grade = '';
            if (i >= 90)
                grade = 'S';
            else if (i >= 80 && i < 90)
                grade = 'A';
            else if (i >= 70 && i < 80)
                grade = 'B';
            else if (i >= 60 && i < 70)
                grade = 'C';
            else if (i >= 55 && i < 60)
                grade = 'D';
            else if (i >= 50 && i < 55)
                grade = 'E';
            else
                grade = 'F';

            var elem = document.getElementById("grade");
            elem.innerHTML = "";
            elem.innerHTML += grade;
        }
    </script>
